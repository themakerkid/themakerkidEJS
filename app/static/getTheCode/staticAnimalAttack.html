<script type="text/javascript" src="https://cdnjs.cloudflare.com/ajax/libs/processing.js/1.3.6/processing.min.js"></script> <!-- processing.js -->
<script>
    var sketchProc = function () { // function to use later in the html code
        with(processingInstance) { // with a processing instance
            size(400, 400); // size of the canvas
            frameRate(30); // how times the draw function is called per second

            var bodyX = 200; // body x position
            var bodyY = 220; // body y position
            var bodyW = 118; // body width
            var bodyH = bodyW / 2; // body height
            var eyeSize = 10; // the size of his eyes

            draw = function () { // draw function
                background(207, 254, 255); // background colour
                fill(240, 209, 36); // fill colour
                rect(bodyX * 1 / 2, 186, 131, 27); // left arm?
                rect(bodyX * 7 / 8, 186, 131, 27); // right arm? Maths (175/200 -> 7/8)
                rect(bodyX * 4 / 5, 249, 30, 89); // left leg? Maths (160/200 -> 16/20 -> 4/5)
                rect(210, 249, 30, 89); // right leg?
                ellipse(bodyX, bodyY, bodyW, 106); // body?
                ellipse(bodyX, bodyY - 70, bodyH, 47); // face?
                ellipse(185, 145, eyeSize, eyeSize); // left eye
                ellipse(215, 145, eyeSize, eyeSize); // right eye
            };
        }
    }
</script>
<script type="application/javascript">
    var canvas = document.getElementById("mycanvas");
    var processingInstance = new Processing(canvas, sketchProc);
</script>
