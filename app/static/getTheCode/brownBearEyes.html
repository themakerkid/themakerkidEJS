<script>
    var sketchProc = function (processingInstance) {
        with(processingInstance) {
            size(400, 400);
            frameRate(30);

            noStroke(); // no borders around our shapes
            var x = 20; // x position
            var y = 350; // y position
            var faceSize = 359; // face size

            // there is some maths calculations in the rest of the program so don't worry if you don't understand them
            // ears
            var earSize = faceSize * 1 / 2; // ear size
            fill(89, 52, 17); // fil colour
            ellipse(x - faceSize * 2 / 5, y - faceSize * 2 / 5, earSize, earSize);
            ellipse(x + faceSize * 2 / 5, y - faceSize * 2 / 5, earSize, earSize);

            // face
            fill(163, 113, 5); // fill colour
            ellipse(x, y, faceSize, faceSize);

            // eyes 
            fill(0, 0, 0); // fill colour
            var eyeSize = faceSize / 8; // eye size
            ellipse(x - faceSize / 4, y - faceSize / 8, eyeSize, eyeSize);
            ellipse(x + faceSize / 4, y - faceSize / 8, eyeSize, eyeSize);



            // nose
            fill(89, 52, 20); // fill colour
            ellipse(x, y + faceSize / 8, faceSize * 4 / 15, faceSize / 5);
        }
    }
</script>
<script type="application/javascript">
    var canvas = document.getElementById("mycanvas");
    var processingInstance = new Processing(canvas, sketchProc);
</script>
