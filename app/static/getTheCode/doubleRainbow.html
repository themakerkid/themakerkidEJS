<script type="text/javascript" src="https://cdnjs.cloudflare.com/ajax/libs/processing.js/1.3.6/processing.min.js"></script> <!-- processing.js -->
<script>
    var sketchProc = function () {
        with(processingInstance) {
            size(400, 400);
            frameRate(30);

            var Rainbow = function (x, y, radius) { // rainbow constructor
            this.x = x;
            this.y = y;
            this.radius = radius;
            };

            var drawRainbow = new Rainbow(200, 200, 300); // rainbow 1

            var drawRainbow2 = new Rainbow(200, 314, 300); // rainbow 2

            var drawRainbow = function (rainbow) { // draws a rainbow
                rainbow.x = 200; // x position
                rainbow.y = 200; // y position
                rainbow.radius = 300; // radius

                noFill();
                strokeWeight(5);
                stroke(255, 0, 0);
                arc(rainbow.x, rainbow.y, rainbow.radius, rainbow.radius, 180, 360); // red
                stroke(250, 200, 0);
                arc(rainbow.x, rainbow.y, rainbow.radius - 10, rainbow.radius - 10, 180, 360); // orange
                stroke(240, 255, 102);
                arc(rainbow.x, rainbow.y, rainbow.radius - 20, rainbow.radius - 20, 180, 360); // yellow
                stroke(29, 255, 13);
                arc(rainbow.x, rainbow.y, rainbow.radius - 30, rainbow.radius - 30, 180, 360); // green
                stroke(122, 155, 255);
                arc(rainbow.x, rainbow.y, rainbow.radius - 40, rainbow.radius - 40, 180, 360); // blue
            };

            drawRainbow(drawRainbow); // draw rainbow 1
            drawRainbow(drawRainbow2); // draw rainbow 2
        }
    }
</script>
<script type="application/javascript">
    var canvas = document.getElementById("mycanvas");
    var processingInstance = new Processing(canvas, sketchProc);
</script>
