<script>
    var sketchProc = function (processingInstance) {
        with(processingInstance) {
            size(400, 350);
            frameRate(30);
            /* This program uses inheritance, don't worry if you don't know what I am talking about. It is very tricky. Here is a link: Click here (very long!) */
            var Flower = function (x, y, height) { // flower constructor
                this.x = x;
                this.y = y;
                this.height = height;
            };

            Flower.prototype.growBy = function (amount) { // growBy method
                this.height += 2 * amount;
            };

            var Tulip = function (x, y, height) { // tulip constructor
                Flower.call(this, x, y, height); // call the Flower constructor
            };

            Tulip.prototype = Object.create(Flower.prototype); // set the prototype

            Tulip.prototype.draw = function () { // draw method
                noStroke();
                fill(16, 122, 12);
                rect(this.x, this.y, 10, -this.height);
                fill(255, 0, 0);
                // petals
                ellipse(this.x + 5, this.y - this.height, 44, 44);
                triangle(this.x - 16, this.y - this.height,
                    this.x + 20, this.y - this.height,
                    this.x - 20, this.y - this.height - 31);
                triangle(this.x - 14, this.y - this.height,
                    this.x + 24, this.y - this.height,
                    this.x + 3, this.y - this.height - 39);
                triangle(this.x + -4, this.y - this.height,
                    this.x + 26, this.y - this.height,
                    this.x + 29, this.y - this.height - 36);
            };


            var Sunflower = function (x, y, height) { // sunflower constructor
                Flower.call(this, x, y, height); // call the Flower constructor
            };

            Sunflower.prototype = Object.create(Flower.prototype); // set the prototype

            Sunflower.prototype.draw = function () { // draw method
                fill(16, 122, 12);

                rect(this.x, this.y, 10, -this.height);

                // petals
                stroke(0, 0, 0);
                fill(255, 221, 0);
                ellipse(this.x - 10, this.y - this.height, 20, 18);
                ellipse(this.x + 5, this.y - this.height - 15, 20, 18);
                ellipse(this.x + 5, this.y - this.height + 15, 20, 18);
                ellipse(this.x + 20, this.y - this.height, 20, 18);
                fill(20, 20, 20);
                ellipse(this.x + 5, this.y - this.height, 20, 20);
            };


            var tulip = new Tulip(38, 390, 150); // make the first tulip
            var sunflower = new Sunflower(186, 390, 100); // make the first sunflower

            var tulip2 = new Tulip(112, 390, 150); // make the second tulip
            var sunflower2 = new Sunflower(243, 390, 150); // make the second sunflower

            var drawScene = function () { // method to draw the scene
                background(207, 250, 255);
                tulip.draw();
                sunflower.draw();
                tulip2.draw();
                sunflower2.draw();

            };

            mouseClicked = function () { // when the mouse is clicked they wilsl grow. Sunflowers grow faster than tulips
                tulip.growBy(10);
                sunflower.growBy(20);
                tulip2.growBy(10);
                sunflower2.growBy(20);
                drawScene();
            };

            drawScene(); // call the method to draw the scene

        }
    };
</script>
<script type="application/javascript">
    var canvas = document.getElementById("mycanvas");
    var processingInstance = new Processing(canvas, sketchProc);
</script>
