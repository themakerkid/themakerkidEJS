<script type="text/javascript" src="https://cdnjs.cloudflare.com/ajax/libs/processing.js/1.3.6/processing.min.js"></script> <!-- processing.js -->
<script>
    var sketchProc = function () {
        with(processingInstance) {
            size(400, 400);
            frameRate(30);
            
            var x = 200;
            var y = 250;

            noStroke();
            fill(30, 204, 91); // a nice froggy green!

            ellipse(x, y, 200, 100); // face
            ellipse(x - 50, y - 50, 40, 40); // left eye socket
            ellipse(x + 50, y - 50, 40, 40); // right eye socket

            fill(255, 255, 255); // for the whites of the eyes!
            ellipse(x - 50, y - 50, 30, 30); // left eyeball
            ellipse(x + 50, y - 50, 30, 30); // right eyeball

            fill(15, 11, 11);
            ellipse(x, y, 136, 67);//mouth

            rect(x - 54, y - 50, 10, 10);
            rect(x + 46, y - 50, 10, 10);

        }
    }
</script>
<script type="application/javascript">
    var canvas = document.getElementById("mycanvas");
    var processingInstance = new Processing(canvas, sketchProc);
</script>
